变量声明

var变量

1. 函数级作用域
2. 作用域提升
  1. 使用非定义的变量时，js会默认创建一个`undefined`的变量
3. 如果不使用`var`声明对象，则该对象在使用后会成为全局变量即`window`对象的一个属性
4. 重复声明不报错 后声明的变量会替代之前的变量

//var变量

//1.函数级作用域
function var_test1(){
    var msg = '123';
    console.log(msg); //123
}
console.log(msg);//error

//2.作用域提升
function var_test2(){
    console.log(msg);//undefined
    var msg = "123";
}

function var_test3(){
    var msg ;
    console.log(msg);//undefined
    msg = "123";
}
//var_test2在编译过程中编译为var_test3

//3.全局声明的变量成为window对象的属性
function var_test4(){
    msg = "global";
    console.log(msg);
}

console.log(window.msg); //global

function var_test5(){
    message = 'def';
    console.log(message);
    var massge;
}

window.message; // error
//var_test5中 函数级作用域对其进行限制导致无法在出现在全局中。

//4.重复声明不报错
var msg = "123";
var msg;

let 变量

1. 块级作用域 `{}` let声明的变量的生命周期只存在于一对 `{}`内
2. let没有变量提升，存在暂时性死区，即在let声明变量前，该变量无法使用
3. 不能重复声明
4. let var 不能混用
5. 全局声明不是window对象


//1.块级作用域
if(true){
 var title = "123";
}
window.title;//123

if(true){
    let content = "abc";
}
window.content;//error

//2.暂时性死区
function let_test1(){
    console.log(msg); //error
    let msg;
    msg = 456;
    console.log(msg); //456
}

//3.不能重复声明
let name;
let name = 123;//error

//4.let var 不能混用
let age;
var age;//error

//4.1 或 
var title;
let title; // error

//5.全局声明不是window对象
let globe = 123;
window.globe;//undefined

const 变量

1. const 一般用来声明常量,同时必须进行初始化
2. 初始化后不能进行修改
3. 不能修改仅限于对常量的引用，如果是对象是可以修改对象内的属性
4. const不能用于会自增的迭代变量，即不能用在普通的循环里，但可以用在`for of` ,`for in` 中
    1. `for of`: 迭代可迭代对象的值
    2. `for in`: 迭代对象的key值


const obj = {a:1};
obj = {} //error 修改为空对象会报错
obj.b = 2; //修改属性是可以的
console.log(obj);//a:1,b:2

//对于for循环，其实是在for循环内即花括号的内部声明小括号里的变量，其作用域只限于花括号内部（每次声明一个新的变量）
for(const key in {a:1,b:2}){
    console.log(key);
}
//a
//b
for(const key of 'abc'){
    console.log(key);
}
//a
//b
//c
